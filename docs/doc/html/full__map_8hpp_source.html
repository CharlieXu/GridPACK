303 /**
304  * Set up the offset arrays that will be used to find the exact location of
305  * each matrix block in the matrix produced by the mapper
306  */
307 void setupOffsetArrays()
308 {
309  int *itmp = new int[p_nNodes];
310  int *jtmp = new int[p_nNodes];
311 
312  int *rptr;
313  int i, idx;
314  int one = 1;
315 
316  // We need to decompose this matrix by rows so that each processor has a
317  // contiguous set of rows. The MatVec indices are set up so that the active
318  // buses on each processor are indexed consecutively. This index can be used
319  // to set up this partition.
320  p_minRowIndex = p_totalBuses;
321  p_maxRowIndex = 0;
322  for (i=0; i<p_nBuses; i++) {
323  if (p_network->getActiveBus(i)) {
324  p_network->getBus(i)->getMatVecIndex(&idx);
325  if (idx > p_maxRowIndex) p_maxRowIndex = idx;
326  if (idx < p_minRowIndex) p_minRowIndex = idx;
327  }
328  }
329 // printf("p[%d] (FullMatrixMap) minRow: %d maxRow: %d\n",p_me,p_minRowIndex,p_maxRowIndex);
330  // Create array to hold information about desired rows
331  int nRows = p_maxRowIndex-p_minRowIndex+1;
332  int *iSizes = new int[nRows];
333  int *jSizes = new int[nRows];
334  GA_Sync();
335  NGA_Get(gaMatBlksI,&p_minRowIndex,&p_maxRowIndex,iSizes,&one);
336  NGA_Get(gaMatBlksJ,&p_minRowIndex,&p_maxRowIndex,jSizes,&one);
337 
338  // Calculate total number of elements associated with row block and column
339  // block associated with this processor
340  int iSize = 0;
341  int jSize = 0;
342  p_maxIBlock = 0;
343  p_maxJBlock = 0;
344  for (i=0; i<nRows; i++) {
345  if (p_maxIBlock < iSizes[i]) p_maxIBlock = iSizes[i];
346  if (p_maxJBlock < jSizes[i]) p_maxJBlock = jSizes[i];
347  if (iSizes[i] > 0) iSize += iSizes[i];
348  if (jSizes[i] > 0) jSize += jSizes[i];
349  if (iSizes[i] == 0 || jSizes[i] == 0) {
350 // printf("p[%d] Sizes[%d] I: %d J: %d\n",p_me,i,iSizes[i],jSizes[i]);
351  }
352  }
353  p_rowBlockSize = iSize;
354  GA_Igop(&p_maxIBlock,one,"max");
355  GA_Igop(&p_maxJBlock,one,"max");
356 
357  for (i = 0; i<p_nNodes; i++) {
358  itmp[i] = 0;
359  jtmp[i] = 0;
360  }
361  itmp[p_me] = iSize;
362  jtmp[p_me] = jSize;
363 // printf("p[%d] (FullMatrixMap) iSize: %d jSize: %d\n",p_me,iSize,jSize);
364 
365  GA_Igop(itmp, p_nNodes, "+");
366  GA_Igop(jtmp, p_nNodes, "+");
367 
368  int offsetArrayISize = 0;
369  int offsetArrayJSize = 0;
370  for (i = 0; i < p_me; i++) {
371  offsetArrayISize += itmp[i];
372  offsetArrayJSize += jtmp[i];
373  }
374 
375  // Calculate matrix dimension
376  p_iDim = 0;
377  p_jDim = 0;
378  for (i=0; i<p_nNodes; i++) {
379  p_iDim += itmp[i];
380  p_jDim += jtmp[i];
381  }
382 // printf("p[%d] (FullMatrixMap) iDim: %d jDim: %d\n",p_me,p_iDim,p_jDim);
383 
384  // Create map array so that offset arrays can be created with a specified
385  // distribution
386  int *offset = new int[p_nNodes];
387  for (i=0; i<p_nNodes; i++) {
388  offset[i] = 0;
389  }
390  offset[p_me] = p_activeBuses;
391 // printf("p[%d] (FullMatrixMap) activeBuses: %d\n",p_me,p_activeBuses);
392  GA_Igop(offset,p_nNodes,"+");
393 
394  int *mapc = new int[p_nNodes];
395  mapc[0]=0;
396  for (i=1; i<p_nNodes; i++) {
397  mapc[i] = mapc[i-1] + offset[i-1];
398  }
399 
400  delete [] offset;
401  delete [] itmp;
402  delete [] jtmp;
403 
404  // Create arrays that will hold final offsets
405  gaOffsetI = GA_Create_handle();
406  GA_Set_data(gaOffsetI, one, &p_totalBuses, C_INT);
407  GA_Set_irreg_distr(gaOffsetI, mapc, &p_nNodes);
408  if (!GA_Allocate(gaOffsetI)) {
409  // TODO: some kind of error
410  }
411  GA_Zero(gaOffsetI);
412 
413  gaOffsetJ = GA_Create_handle();
414  GA_Set_data(gaOffsetJ, one, &p_totalBuses, C_INT);
415  GA_Set_irreg_distr(gaOffsetJ, mapc, &p_nNodes);
416  if (!GA_Allocate(gaOffsetJ)) {
417  // TODO: some kind of error
418  }
419  GA_Zero(gaOffsetJ);
420 
421  // Evaluate offsets for this processor
422  int *iOffsets = new int[nRows];
423  int *jOffsets = new int[nRows];
424  iOffsets[0] = offsetArrayISize;
425  jOffsets[0] = offsetArrayJSize;
426  for (i=1; i<nRows; i++) {
427  iOffsets[i] = iOffsets[i-1] + iSizes[i-1];
428  jOffsets[i] = jOffsets[i-1] + jSizes[i-1];
429  }
430 
431  // Put offsets into global arrays
432  if (nRows > 0) {
433  NGA_Put(gaOffsetI,&p_minRowIndex,&p_maxRowIndex,iOffsets,&one);
434  NGA_Put(gaOffsetJ,&p_minRowIndex,&p_maxRowIndex,jOffsets,&one);
435  }
436  GA_Sync();
437 
438  // Clean up arrays that are no longer needed
439  GA_Destroy(gaMatBlksI);
440  GA_Destroy(gaMatBlksJ);
441 
442  delete [] mapc;
443  delete [] iSizes;
444  delete [] jSizes;
445  delete [] iOffsets;
446  delete [] jOffsets;
447 }

507  }

524  } else {

526  }
527  icnt++;
528  }
529  }

531  jcnt++;
532  }
533  }
534  }
