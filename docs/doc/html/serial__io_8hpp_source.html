78  int nBus = p_network->numBuses();
79  char string[p_size];
80  int nwrites = 0;
81  int i;
82  int one = 1;
83 
84  // Count up total strings being written from this processor
85  for (i=0; i<nBus; i++) {
86  if (p_network->getActiveBus(i) &&
87  p_network->getBus(i)->serialWrite(string,signal)) nwrites++;
88  }
89 
90  // Set up buffers to scatter strings to global buffer
91  int **index;
92  index = new int*[nwrites];
93  int ones[nwrites];
94  char strbuf[nwrites*p_size];
95  char *ptr = strbuf;
96  nwrites = 0;
97  for (i=0; i<nBus; i++) {
98  if (p_network->getActiveBus(i) &&
99  p_network->getBus(i)->serialWrite(ptr,signal)) {
100  index[nwrites] = new int;
101  *(index[nwrites]) = p_network->getGlobalBusIndex(i);
102  ones[nwrites] = 1;
103  nwrites++;
104  ptr += p_size;
105  }
106  }
107 
108  // Scatter data to global buffer and set mask array
109  GA_Zero(p_maskGA);
110  if (nwrites > 1) {
111  NGA_Scatter(p_stringGA,strbuf,index,nwrites);
112  NGA_Scatter(p_maskGA,ones,index,nwrites);
113  }
114  GA_Sync();
115  for (i=0; i<nwrites; i++) {
116  delete index[i];
117  }
118  delete [] index;
119

121  // from each successive processor and writes it to standard out
122  if (GA_Nodeid() == 0) {

135  nwrites++;

142  nwrites = 0;
