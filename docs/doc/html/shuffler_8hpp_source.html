36 /// A functor to redistribute things in vectors on multiple processes
37 /**
38  *
39  * Each process starts with a (possibly empty) vector of things and a
40  * vector of equal size that containing a destination process for each
41  * thing. After execution, each process will contain a vector of the
42  * things assigned to it.
43  *
44  * This uses blocking send/receive.
45  *
46  * The things redistributed must be copy constructable and serializable.
47  *
48  */
49 
50 template <typename Thing, typename I = int>
51 struct Shuffler {
52 
53  // Thing must be copyable and serializable and have a constructor
54  // without arguments -- should test for that
55 
56  typedef std::vector<Thing> ThingVector;
57  typedef I Index;
58  typedef std::vector<Index> IndexVector;
59  typedef std::pair<Index, Index> iPair;
60  typedef std::vector<iPair> iPairVector;
61 
62  /// Redistribute and get the Things assigned to the local process
63  void operator()(const boost::mpi::communicator& comm,
64  ThingVector& locthings, const IndexVector& destproc)
65  {
66  BOOST_ASSERT(locthings.size() == destproc.size());
67 
68  size_t nthings(0);
69  all_reduce(comm, locthings.size(), nthings, std::plus<size_t>());
70  if (nthings <= 0) return;
71 
72  // save the original list of local things
73 
74  ThingVector tvect;
75  tvect.reserve(locthings.size());
76  std::copy(locthings.begin(), locthings.end(),
77  std::back_inserter(tvect));
78  locthings.clear();
79 
80  int msgid(0); // unique MPI message id
81 
82  // Work on each processors list of things separately
83 
84  iPairVector srcdest;
85  for (int p = 0; p < comm.size(); ++p) {
86 
87  // build a set of source-destination pairs for each thing in p's
88  // local list
89 
90  srcdest.clear();
91  if (comm.rank() == p) {
92  for (typename IndexVector::const_iterator i = destproc.begin();
93  i != destproc.end(); ++i) {
94  iPair apair(p, *i);
95  srcdest.push_back(apair);
96  }
97  }
98 
99  // all processes get the source-destination pairs
100 
101  broadcast(comm, srcdest, p);
102 
103  // all processes go through the source-destination pairs and
104  // build a message request list
105 
106  size_t locidx(0);
107  for (typename iPairVector::iterator pd = srcdest.begin();
108  pd != srcdest.end(); ++pd, ++msgid) {
109  int src(pd->first);
110  int dest(pd->second);
111  if (comm.rank() == src) {
112  if (src == dest) {
113  locthings.push_back(tvect[locidx]);
114  // std::cout << src << ": kept " << locthings.back() << " @ "
115  // << locthings.size() - 1 << std::endl;
116  } else {
117  comm.send(dest, msgid, tvect[locidx]);
118  // std::cout << src << ": msg " << msgid << ": sent " <<
119  // tvect[locidx] << std::endl;
120  }
121  locidx += 1;
122  } else if (comm.rank() == dest) {
123  Thing bogus;
124  locthings.push_back(bogus);
125  comm.recv(src, msgid, locthings.back());
126  // std::cout << dest << ": msg " << msgid
127  // << ": got @ " << locthings.size() - 1 << " \""
128  // << locthings.back() << "\"" << std::endl;
129  }
130  comm.barrier();
131  }
132  }
133 
134  // this needs to be here in case a processor was not involved in
135  // any transfer
136 
137  comm.barrier();
138  }

140 };
