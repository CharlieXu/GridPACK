197  /**
198  * Return the size of the buffer needed for data exchanges. Note that this
199  * must be the same size for all bus and all branch objects (branch buffers
200  * do not need to be the same size as bus buffers), even if all objects
201  * do not require the same parameters. Thus, the buffer must be big enough
202  * to exchange all variables that an object might need, even if individual
203  * objects don't need all the variables
204  */
205  virtual int getXCBufSize(void);
206 
207  /**
208  * Assign the location of the data exchange buffer. These buffers are
209  * allocated and deallocated by the network
210  * @param buf: void pointer to exchange buffer
211  */
212  virtual void setXCBuf(void *buf);
213 
214  /**
215  * Set an internal variable that can be used to control the behavior of the
216  * component. This function doesn't need to be implemented, but if needed,
217  * it can be used to change the behavior of the network in different phases
218  * of the calculation. For example, if a different matrix needs to be
219  * generated at different times, the mode of the calculation can changed to
220  * get different values from the MatVecInterface functions
221  * @param mode: integer indicating which mode should be used
222  */
223  virtual void setMode(int mode);
224 
225  /**
226  * Copy a string for output into buffer. The behavior of this method can be
227  * altered by inputting different values for the signal string
228  * @param string: buffer containing string to be written to output
229  * @param signal: string to control behavior of routine (e.g. what
230  * properties to write
231  * @return: true if component is writing a contribution, false otherwise
232  */
233  virtual bool serialWrite(char *string, char *signal = NULL);
234 
235  protected:
236  /**
237  * A buffer that can be used for exchanging component data. This is
238  * allocated by the network based on an inquiry to the getXCBusSize method
239  */
240  void *p_XCBuf;
241 
242  /**
243  * Size (in bytes) of buffer p_XCBuf
244  */
245  int p_XCBufSize;
246 
247  /**
248  * Current mode
249  */
250  int p_mode;
251 
252  private:
253 
254  friend class boost::serialization::access;
255 
256  /// Serialization "pack" routine
257  template<class Archive>
258  void save(Archive & ar, const unsigned int version) const
259  {
260  // p_XCBuf and p_XCBufSize are managed somewhere else; they will
261  // have to be initialized
