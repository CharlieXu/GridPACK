192 /**
193  * Set up the offset arrays that will be used to find the exact location of
194  * each vector block in the vector produced by the mapper
195  */
196 void setupOffsetArrays()
197 {
198  int *itmp = new int[p_nNodes];
199 
200  int *rptr;
201  int i, idx;
202  int one = 1;
203 
204  // We need to decompose this vector so that each processor has a contiguous
205  // set of elements. The MatVec indices are set up so that the active
206  // buses on each processor are indexed consecutively. This index can be used
207  // to set up this partition.
208  p_minRowIndex = p_totalBuses;
209  p_maxRowIndex = 0;
210  for (i=0; i<p_nBuses; i++) {
211  if (p_network->getActiveBus(i)) {
212  p_network->getBus(i)->getMatVecIndex(&idx);
213  if (idx > p_maxRowIndex) p_maxRowIndex = idx;
214  if (idx < p_minRowIndex) p_minRowIndex = idx;
215  }
216  }
217  // Create array to hold information about desired elements
218  int nRows = p_maxRowIndex-p_minRowIndex+1;
219  int *iSizes = new int[nRows];
220  GA_Sync();
221  NGA_Get(gaVecBlksI,&p_minRowIndex,&p_maxRowIndex,iSizes,&one);
222 
223  // Calculate total number of elements associated with row block and column
224  // block associated with this processor
225  int iSize = 0;
226  p_maxIBlock = 0;
227  for (i=0; i<nRows; i++) {
228  if (p_maxIBlock < iSizes[i]) p_maxIBlock = iSizes[i];
229  if (iSizes[i] > 0) iSize += iSizes[i];
230  }
231  p_rowBlockSize = iSize;
232  GA_Igop(&p_maxIBlock,one,"max");
233 
234  for (i = 0; i<p_nNodes; i++) {
235  itmp[i] = 0;
236  }
237  itmp[p_me] = iSize;
238 
239  GA_Igop(itmp, p_nNodes, "+");
240 
241  int offsetArrayISize = 0;
242  for (i = 0; i < p_me; i++) {
243  offsetArrayISize += itmp[i];
244  }
245 
246  // Calculate vector length
247  int p_iDim = 0;
248  for (i=0; i<p_nNodes; i++) {
249  p_iDim += itmp[i];
250  }
251 
252  // Create map array so that offset arrays can be created with a specified
253  // distribution
254  int *offset = new int[p_nNodes];
255  for (i=0; i<p_nNodes; i++) {
256  offset[i] = 0;
257  }
258  offset[p_me] = p_activeBuses;
259  GA_Igop(offset,p_nNodes,"+");
260 
261  int *mapc = new int[p_nNodes];
262  mapc[0]=0;
263  for (i=1; i<p_nNodes; i++) {
264  mapc[i] = mapc[i-1] + offset[i-1];
265  }
266 
267  delete [] offset;
268  delete [] itmp;
269 
270  // Create arrays that will hold final offsets
271  gaOffsetI = GA_Create_handle();
272  GA_Set_data(gaOffsetI, one, &p_totalBuses, C_INT);
273  GA_Set_irreg_distr(gaOffsetI, mapc, &p_nNodes);
274  if (!GA_Allocate(gaOffsetI)) {
275  // TODO: some kind of error
276  }
277  GA_Zero(gaOffsetI);
278 
279  // Evaluate offsets for this processor
280  int *iOffsets = new int[nRows];
281  iOffsets[0] = offsetArrayISize;
282  for (i=1; i<nRows; i++) {
283  iOffsets[i] = iOffsets[i-1] + iSizes[i-1];
284  }
285 
286  // Put offsets into global arrays
287  if (nRows > 0) {
288  NGA_Put(gaOffsetI,&p_minRowIndex,&p_maxRowIndex,iOffsets,&one);
289  }
290  GA_Sync();
291 
292  delete [] mapc;
293  delete [] iSizes;
294  delete [] iOffsets;
295 }

325 /**
326  * Add block contributions from buses to vector
327  * @param vector: vector to which contributions are added
328  * @param flag: flag to distinguish new vector (true) from existing vector * (false)
329  */
330 void loadBusData(boost::shared_ptr<gridpack::math::Vector> &vector, bool flag)
331 {
332  int i,idx,isize,icnt;
333  int **indices = new int*[p_busContribution];
334  icnt = 0;
335  for (i=0; i<p_nBuses; i++) {
336  if (p_network->getActiveBus(i)) {
337  if (p_network->getBus(i)->vectorSize(&isize)) {
338  indices[icnt] = new int;
339  p_network->getBus(i)->getMatVecIndex(&idx);
340  *(indices[icnt]) = idx;
341  icnt++;
342  }
343  }
344  }
345  if (icnt != p_busContribution) {
346  // TODO: some kind of error
347  }
348 
349  // Gather vector offsets
350  int *offsets = new int[p_busContribution];
351  if (p_busContribution > 0) {
352  NGA_Gather(gaOffsetI,offsets,indices,p_busContribution);
353  }
354 
355  // Add vector elements
356  ComplexType *values = new ComplexType[p_maxIBlock];
357  int j;
358  int jcnt = 0;
359  for (i=0; i<p_nBuses; i++) {
360  if (p_network->getActiveBus(i)) {
361  if (p_network->getBus(i)->vectorSize(&isize)) {
362  p_network->getBus(i)->vectorValues(values);
363  icnt = 0;
364  for (j=0; j<isize; j++) {
365  idx = offsets[jcnt] + j;
366  if (flag) {
367  vector->add_element(idx, values[icnt]);
368  } else {
369  vector->set_element(idx, values[icnt]);
370  }
371  icnt++;
372  }
373  delete indices[jcnt];
374  jcnt++;
375  }
376  }
377  }
378 
379  // Clean up arrays
380  delete [] indices;
381  delete [] offsets;
382  delete [] values;
383 }

385 /**
386  * Push data from vector onto buses. Vector must be created with the
387  * mapToVector method using the same BusVectorMap
388  * @param vector: vector containing data to be pushed to buses
389  */
390 void mapToBus(boost::shared_ptr<gridpack::math::Vector> &vector)
391 {
392  int minVecIndex, maxVecIndex;
393  vector->local_index_range(minVecIndex, maxVecIndex);
394 
395  // Assume that row partitioning is working correctly
396  int nRows = p_maxRowIndex - p_minRowIndex + 1;
397  int *sizes = new int[nRows];
398  int *offsets = new int[nRows];
399  int one = 1;
400  NGA_Get(gaVecBlksI,&p_minRowIndex,&p_maxRowIndex,sizes,&one);
401  NGA_Get(gaOffsetI,&p_minRowIndex,&p_maxRowIndex,offsets,&one);
402  ComplexType *values = new ComplexType[p_maxIBlock];
403  int i, j, idx, size, offset;
404  for (i=0; i<p_nBuses; i++) {
405  if (p_network->getActiveBus(i)) {
406  if (p_network->getBus(i)->vectorSize(&size)) {
407  p_network->getBus(i)->getMatVecIndex(&idx);
408  idx = idx - p_minRowIndex;
409  size = sizes[idx];
410  offset = offsets[idx];
411  for (j=0; j<size; j++) {
412  vector->get_element(offset+j,values[j]);
413  }
414  p_network->getBus(i)->setValues(values);
415  }
416  }
417  }
418  delete [] sizes;
419  delete [] offsets;
420 }
